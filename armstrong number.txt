#include<stdio.h>
#include<math.h>
int main()
{
    int num,digit=0,i,sum=0,temp;
    printf("enter a num value:");
    scanf("%d",&num);
    for(i=num;i>0;i=i/10)
    digit++;
    temp=num;
    while(num>0)
    {
    sum=sum+pow(num%10,digit);
    num=num/10;
    }
    if(temp==sum){
        printf("armstrong number");
    }else{
        printf("not a armstrong number");
    }
    return 0;
}
